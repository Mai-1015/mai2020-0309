
##程序说明
代码主函数是一个菜单，使用switch case用来选择哪一个排序  
然后就是各个的排序函数和一个退出函数

* 冒泡排序函数  maopao  使用了for循环的嵌套来实现
* 桶排序函数    tong    使用了二维数组，将100个数分为100个数组，对应装桶
* 快速排序函数  kuaisu  使用了递归思想，找准基准点然后两边分别排放
* 归并排序函数  guibing 也使用了递归思想，先拆分数组，再排序治理数组
* 堆排序函数  dui  也使用了递归思想，先确定好支点，再进行排序交换，  
  再将最大和最小交换，移除最大，反复进行
* 计数排序函数 jishu1 使用了for循环的嵌套，大于多少个数，就排在第几，注意重复的数
* 基数排序函数 jishu2 使用了二维数组，和桶排序类似，但是不用100个数组这么多

int Partition(int arr[], int low, int high)
{
    int base = arr[low];
    while(low < high)
    {
        while(low < high && arr[high] >= base)
        {
            high --;
        }
    	 int t;
   		 t = arr[low];
    	 arr[low] = arr[high];
   		 arr[high] = t;
        while(low < high && arr[low] <= base)
        {
            low ++;
        }
   		 t = arr[low];
    	 arr[low] = arr[high];
   		 arr[high] = t;
    }
    return low;
}
 
void QuickSort(int arr[], int low, int high)
{
    if(low < high)
    {
        int base = Partition(arr, low, high);
        QuickSort(arr, low, base - 1);
        QuickSort(arr, base + 1, high);
    }
}

void zhi(int a[],int b[], int left, int mid, int right)
{//实现“治”
    int i = left, j=mid+1, k = left;
    while(i!=mid+1 && j!=right+1)
    {
        if(a[i] > a[j])
            b[k++] = a[j++];
        else
            b[k++] = a[i++];
    }
    while(i != mid+1) 
        {
			b[k++] = a[i++];
		}
    while(j != right+1)
        {
			b[k++] = a[j++];
		}
    for(i=left; i<=right; i++)
        {
			a[i] = b[i];
		}
}

void fen(int a[],int b[],int left,int right)
{//实现“分”
	int mid;
	if(left < right)
	{
		mid =(left+right)/2;
		fen(a, b,left,mid);
		fen(a, b,mid+1,right);
		zhi(a, b,left,mid,right);
	}
}

void dui1(int a[],int n,int i)
{
	if(i>=n)
	{
		return;
	}
	int c1 = 2*i+1;
	int c2 = 2*i+2;
	int max = i;
	if(c1<n&&a[c1]>a[max])
	{
		max = c1;
	}
	if(c2<n&&a[c2]>a[max])
	{
		max = c2;
	}
	if(max!=i)
	{
		int t;
		t=a[i];
		a[i]=a[max];
		a[max]=t;
		dui1(a,n,max);
	}		
}
 
void dui2(int a[],int n)
{
	int last = n-1;
	int now = (last-1)/2;
	int i;
	for(i=now;i>=0;i--)
	{
		dui1(a,n,i);
	}
}
 
 
void dui3(int a[],int n)
{
	dui2(a,n);
	int i;
	for(i=n-1;i>=0;i--)
	{
		int t;
		t=a[i];
		a[i]=a[0];
		a[0]=t;	
		dui1(a,i,0);
	}
}
